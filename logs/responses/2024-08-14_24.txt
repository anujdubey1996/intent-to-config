
Create a Kubernetes deployment workload configuration file based on the provided intent.

Intent: Deploy two versions of a GraphQL service in the graphql-example namespace. Ensure that both deployments are configured with readiness probes and environment variables sourced from ConfigMaps and Secrets.

```yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  namespace: graphql-example
  name: graphql-service-v1
spec:
  replicas: 3
  selector:
    matchLabels:
      app: graphql-service
      version: v1
  template:
    metadata:
      labels:
        app: graphql-service
        version: v1
    spec:
      containers:
      - name: graphql-service
        image: your-image/graphql-service:v1
        ports:
        - containerPort: 4000
        envFrom:
        - secretRef:
            name: graphql-service-v1- secrets
        readinessProbe:
          httpGet:
            path: /health
            port: 4000
      imagePullSecrets:
      - name: your-image-pull-secret
---
apiVersion: apps/v1
kind: Deployment
metadata:
  namespace: graphql-example
  name: graphql-service-v2
spec:
  replicas: 3
  selector:
    matchLabels:
      app: graphql-service
      version: v2
  template:
    metadata:
      labels:
        app: graphql-service
        version: v2
    spec:
      containers:
      - name: graphql-service
        image: your-image/graphql-service:v2
        ports:
        - containerPort: 4001
        envFrom:
        - configMapRef:
            name: graphql-service-v2-configmap
        readinessProbe:
          httpGet:
            path: /health
            port: 4001
      imagePullSecrets:
      - name: your-image-pull-secret
```
